# Slack App 설정
# https://api.slack.com/apps 에서 앱 생성 후 아래 값들을 입력하세요
# 이 파일을 .env로 복사하여 사용하세요

# Slack Bot Token (xoxb-로 시작)
SLACK_BOT_TOKEN=xoxb-your-bot-token

# Slack App Token (xapp-로 시작, Socket Mode 사용 시 필요)
SLACK_APP_TOKEN=xapp-your-app-token

# Slack Signing Secret (앱 설정의 Basic Information에서 확인)
SLACK_SIGNING_SECRET=your-signing-secret

# AI Provider 설정 (claude 또는 gemini)
AI_PROVIDER=gemini

# Claude API Key (AI_PROVIDER가 claude인 경우)
ANTHROPIC_API_KEY=your-anthropic-api-key

# Gemini API Key (AI_PROVIDER가 gemini인 경우)
GEMINI_API_KEY=your-gemini-api-key

# Jira 설정 (선택사항)
JIRA_BASE_URL=https://your-domain.atlassian.net
JIRA_EMAIL=your-email@example.com
JIRA_API_TOKEN=your-jira-api-token
JIRA_PROJECT_KEY=YOUR_PROJECT

# 서버 포트 (HTTP 모드 사용 시)
PORT=3001
